## 1 keys() :-  is a dictionary that Returns a list of keys
# example:-
derby = {"manchster", "lancshite", "el clasico"} # a variable is specified in a list
print(derby.keys()) # keys will be shown

## 2 values() :- is a dictionary that is a dictionary that Returns a list of values.
# example:-
student {"uni": "AASTU", "year": "3"} #  a variable is specified in a list
print(student.values()) # a dictionary called values is used

## 3 items() :- is a dictionary that  Returns a list of key-value pairs
# example:-
student {"uni": "AASTU", "year": "3"} #  a variable is specified in a list
# Creating a dictionary with fruit prices
fruit_prices = {
    "apple": 1.5,
    "banana": 0.75,
    "orange": 1.2
}

# Accessing values
print(fruit_prices["banana"])  # Output: 0.75

# Adding a new key-value pair
fruit_prices["grape"] = 2.0

# Updating a value
fruit_prices["apple"] = 1.8

# Removing a key-value pair
del fruit_prices["orange"]

# Printing the updated dictionary
print(fruit_prices)



## 4 get() :- is a dictionary that returns the value for the specified key if key is in
 dictionary.
# example:-
student {"uni": "AASTU", "year": "3"} #  a variable is specified in a list
print(student.get("uni)) # this will print the uni called AASTU
